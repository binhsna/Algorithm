#include<stdio.h>
#include<conio.h>
#include<string.h>

void Hoan_Vi(char &x, char &y)
{
	char temp = x;
	x = y;
	y = temp;
}

// xây dựng lại hàm đảo ngược chuỗi: strrev
void STRREV(char s[])
{
	int dodai = strlen(s);
	for(int i = 0; i < (dodai / 2); i++)
	{
		// đổi chỗ cặp kí tự cho nhau theo công thức: s[i] hoán vị s[dodai - i - 1]
		Hoan_Vi(s[i], s[dodai - i - 1]);
	}
}

int main()
{
	char s[100];

	printf("\nNhap chuoi: ");
	gets(s);
	printf("\nChuoi nhan duoc: %s", s);
	STRREV(s);
	printf("\nChuoi sau khi dao nguoc: %s", s);


	getch();
	return 0;
}
